2.在项目下创建文件夹和文件

文件内容

  1. index.css

    p{
    font-size: 24px;
    padding:0 100px;
    color:blue;
}
p:nth-of-type(2) {
    font-size: 30px;

p:nth-of-type(3) {
    color: red;

    text-align: right;

  2. index.json

    {
  "name":"hello webpack",
  "content":"this is my first demo",
  "start":"Ready Go!"
}
  3. createdom.js
var message=require('./../data/index.json');
module.exports = function() {
    var greet=document.createElement('div');
    greet.innerHTML="<p>"+message.name+"</p>"+"<p>"+message.content+"</p>"+"<p>"+message.start+"</p>";
    return greet;
};

  1. entry.js

    require('./../css/index.css');
var createdom = require('./createdom.js');
document.getElementById('app').appendChild(createdom());

3.命令操作

在webstorm的Terminal窗口执行如下命令：

命令行

a). 生成依赖文件package.json
//生成package.json

1. npm  init -y

b). 依次安装相关依赖

  1. npm intsall webpack

  2. npm install –save-dev webpack

  3. npm install –save-dev css-loader

  4. npm install –save-dev style-loader

  5. npm install –save-dev json-loader

  6. npm install –save-dev babel-core

  7. npm install –save-dev babel-loader

  8. npm install –save-dev webpack-dev-server

4.配置webpack.config.js文件

在项目根路径下创建webpack.config.js文件，文件内容为：

var webpack = require('webpack');
module.exports = {
    //2、进出口文件配置
    entry:'./jsproject/entry.js',//指定的入口文件,“__dirname”是node.js中的一个全局变量，它指向当前执行脚本所在的目录
    output: {//输出
        path: __dirname+'/index',//输出路径
        filename: 'bundle.js'//输出文件名
    },
    module: {//在配置文件里添加加载器说明，指明每种文件需要什么加载器处理
        loaders: [
            {//json加载器
                test: /\.json$/,
                loader: "json-loader"//注意-loader不能省略，网上说能省略，经测试编译会报错
            },
            {//5、编译es6配置
                test:/\.js$/,
                exclude:/node_modules/,
                loader:'babel-loader'
            },
            {//3、CSS-loader
                test:/\.css$/,
                loader:'style-loader!css-loader'//添加对样式表的处理
            }

        ]
    },
    //4、服务器依赖包配置
    devServer: {//注意：网上很多都有colors属性，但是实际上的webpack2.x已经不支持该属性了
        contentBase: "./index",//本地服务器所加载的页面所在的目录
        historyApiFallback: true,//不跳转
        inline: true//实时刷新
        //hot：true,//不要书写该属性，否则浏览器无法自动更新
        //publicPath："/asses/",//设置该属性后，webpack-dev-server会相对于该路径
    },
    plugins:[]//插件
}

5.启动服务

webpack

webpack-dev-server
